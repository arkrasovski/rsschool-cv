```
exports.min = function min(array) {
    if (array == undefined || array.length == 0) {
        return 0;
    }
    let minEl = array[0];
    for (let i = 1; i < array.length; i++) {
        if (array[i] < minEl) {
            minEl = array[i];
        }
    }
    return minEl;
};
```
